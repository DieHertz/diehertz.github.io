import numpy as np
import matplotlib.pyplot as plot
from itertools import groupby

def get_latencies(filename):
    with open(filename) as file:
        content = file.readlines()

    events = [x.strip().split(',') for x in content[1::]]
    events = sorted(events, key=lambda event: float(event[0]))
    
    def fltr(event):
        (timestamp, protocol, value) = event
        return value if protocol == 'sbus' else protocol

    events = [ next(group) for _, group in groupby(events, fltr)]

    last_out_value = None
    pwm_queue = []
    latencies = []
    frame_stamps = []

    first = True
    for timestamp, protocol, value in events:
        if protocol == 'PWM':
            if not first:
                pwm_queue.append((timestamp, protocol, value))
            else:
                first = False

            # if len(pwm_queue) > 1:
                # print('Multiple input changes with no change in output', pwm_queue)
        elif protocol == 'sbus':
            out_value = int(value, 0)

            #
            for idx, pwm in reversed(list(enumerate(pwm_queue))):
                switch_time = float(pwm[0])
                switch_value = 0x713 if int(pwm[2], 0) == 0 else 0xAC

                # ensure value has actually changed
                if last_out_value:
                    value_changed = abs(last_out_value - out_value) > 1000
                    value_is_correct = abs(switch_value - out_value) < 1000

                    if value_changed:
                        if value_is_correct:
                            frame_start = float(timestamp)

                            latency = frame_start - switch_time
                            latencies.append(1e3 * latency)
                            frame_stamps.append(frame_start)

                            pwm_queue.pop(idx)
                            # print latency * 1e3, switch_time, frame_start
                            break
                        else:
                            print('Output frame', (timestamp, protocol, value), 'is lagging past an update in input')

            last_out_value = out_value

    latency_avg = np.mean(latencies)
    latency_std = np.std(latencies)
    latency_q = [ np.percentile(latencies, q) for q in [0, 25, 50, 75, 95, 100]]
    latency_min = latency_q[0]
    latency_max = latency_q[-1]

    print('mean\t', '%.4f' % latency_avg, ' ms')
    print('median\t', '%.4f' % latency_q[2], ' ms')
    print('std\t', '%.4f' % latency_std, ' ms')
    print('min\t', '%.4f' % latency_min, ' ms')
    print('max\t', '%.4f' % latency_max, ' ms')

    import matplotlib.pyplot as plot

    weights = 100 * np.ones_like(latencies) / float(len(latencies))

    figure = plot.figure()

    plot.hist(x=latencies,bins=int(latency_max - latency_min),weights=weights)
    plot.xlabel('latency [ms]')
    plot.ylabel('density [%]')
    plot.show()

    plot.plot(frame_stamps, latencies)
    plot.xlabel('time [s]')
    plot.ylabel('latency [ms]')
    plot.show()
    
    return (latency_avg, latency_std, latency_q)
